<button onclick="clearData()">リセット</button>

<pre id="output"></pre>

<script>
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('sw.js');
}

let data = JSON.parse(localStorage.getItem("records") || "[]");

function now(){
  return new Date().toLocaleTimeString();
}

function save(){
  localStorage.setItem("records", JSON.stringify(data));
}

function render(){
  document.getElementById("timeline").innerHTML =
    data.map(d => `<div>${d}</div>`).join("");
}

function addIV(){
  data.push(now()+" 静脈路確保（成功）");
  save(); render();
}

function addAd(){
  data.push(now()+" アドレナリン1mg");
  save(); render();
}

function addLT(){
  data.push(now()+" LT挿入（成功）");
  save(); render();
}

function addTube(){
  data.push(now()+" 挿管（成功）");
  save(); render();
}

function addGlu(){
  data.push(now()+" ブドウ糖投与");
  save(); render();
}

function generate(){
  let text = data.join("。\n") + "。";
  document.getElementById("output").innerText = text;
}

function pdf(){
  let w = window.open();
  w.document.write("<pre>"+document.getElementById("output").innerText+"</pre>");
  w.print();
}

function clearData(){
  if(confirm("消す？")){
    data=[]; save(); render();
  }
}

render();
</script>
